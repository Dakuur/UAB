def graficar(funcion_sympy, variable):
    # Derivar la funci
    primera_derivada = sp.diff(funcion_sympy, variable)
    segona_derivada = sp.diff(primera_derivada, variable)
    
    # Trobar els punts crtics
    punts_critics = sp.solve(primera_derivada, variable)
    
    # Crear un rang de valors per a graficar la funci
    x_vals = np.linspace(-10, 10, 400)
    y_vals = [funcion_sympy.subs(variable, val) for val in x_vals]
    
    # Graficar la funci
    plt.plot(x_vals, y_vals, label='Funci')
    
    # Avaluar els punts crtics i determinar si sn mnims o mxims
    for punt in punts_critics:
        valor_segona_derivada = segona_derivada.subs(variable, punt)
        if valor_segona_derivada > 0:
            # Mnim
            plt.plot(punt, funcion_sympy.subs(variable, punt), 'go', label='Mnim')
        elif valor_segona_derivada < 0:
            # Mxim
            plt.plot(punt, funcion_sympy.subs(variable, punt), 'ro', label='Mxim')
    
    # Mostrar el grfic
    plt.xlabel('x')
    plt.ylabel('f(x)')
    plt.legend()
    plt.title('Grfic de la funci amb punts crtics')
    plt.grid(True)
    plt.show()

from sympy import symbols, diff, solve

# Definir la variable i la funci
x = symbols('x')
f = (x-2)**2

# Derivades de primer i segon ordre
df = diff(f, x)
d2f = diff(df, x)

# Restricci
def constraint(x):
    return x >= 1

# Trobar punts crtics
critical_points = solve(df, x)

# Analitzar punts crtics
for point in critical_points:
    if not constraint(point): # Si no compleix la restricci,
        continue             # no l'avaluem
    hessian_value = d2f.subs(x, point)  # Utilitzem la segona derivada per a una funci d'una variable
    print(f"\nPunt crtic: {point}")
    print(f"Valor de la segona derivada: {hessian_value}")
    if hessian_value > 0:
        print("s un mnim local.")
    elif hessian_value < 0:
        print("s un mxim local.")
    else:
        print("s un punt de sella.")
